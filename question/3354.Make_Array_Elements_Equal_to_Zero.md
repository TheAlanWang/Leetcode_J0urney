## Method 01 - Prefix
### Approach
prefix
### Code
```python
class Solution:
    def countValidSelections(self, nums: List[int]) -> int:
        n = len(nums)
        left, right = 0, sum(nums)
        count = 0
        for idx in range(n):
            if nums[idx] == 0:
                if 0 <= left - right <= 1:
                    count += 1
                if 0 <= right - left <= 1:
                    count += 1
            else:
                left += nums[idx]
                right -= nums[idx]
        
        return count
```
## Method 02 - Brute Force
### Complexity
- Time Complexity: O(n)
- Space Complexity: O(n)
### Code
```python
class Solution:
    def countValidSelections(self, nums: List[int]) -> int:
        n = len(nums)
        count = 0
        def traversal(idx, moveleft):
            arr = nums[:]
            remaining = sum(arr)
            while True:
                if remaining == 0:
                    return True
                if not(0 <= idx <= n - 1):
                    return False
            
                if arr[idx] == 0:
                    if moveleft:
                        idx -= 1
                    else:
                        idx += 1
                else:
                    arr[idx] -= 1
                    remaining -= 1
                    
                    moveleft = not moveleft
                    if moveleft:
                        idx -= 1
                    else:
                        idx += 1

                    
        for idx in range(n):
            if traversal(idx, True):
                count += 1
            if traversal(idx, False):
                count += 1
        
        return count
```
