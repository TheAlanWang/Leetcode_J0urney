class Solution:
    def canPartition(self, nums):
        total = sum(nums)
        if total % 2 == 1:
            return False
        target = total // 2
        # quick prune
        if max(nums) > target:
            return False

        dp = [False] * (target + 1)
        dp[0] = True

        for x in nums:
            for s in range(target, x - 1, -1):
                dp[s] = dp[s] or dp[s - x]
        return dp[target]
