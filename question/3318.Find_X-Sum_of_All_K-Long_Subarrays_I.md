Link: [3318. Find X-Sum of All K-Long Subarrays I](https://leetcode.com/problems/find-x-sum-of-all-k-long-subarrays-i/)
## Method 1 - sorted
### Approach
1. Use `Counter` to record the frequency within the current sliding window
2. Sorted items by frequency, descending
3. Sum `freq[:x]`, take the top x elements from sorted list
#### Complexity
- Time Complexity: $O(N*K*logk)$
- Space Complexity: $O(k)$
```python
class Solution:
    def findXSum(self, nums: List[int], k: int, x: int) -> List[int]:
        n = len(nums)
        ans = list()
        for i in range(n - k + 1):
            cnt = Counter(nums[i : i + k])
            freq = sorted(cnt.items(), key=lambda item: (-item[1], -item[0]))
            xsum = sum(key * value for key, value in freq[:x])
            ans.append(xsum)
        return ans
```
## Method 2 - dictionary + heap
#### Complexity
- Time Complexity: $O(N*K*logk)$
- Space Complexity: $O(k)$

class Solution:
    def findXSum(self, nums: List[int], k: int, x: int) -> List[int]:

        def findFreq(left, right):
            temp_num = nums[left: right]
            freq_dic = Counter(temp_num)

            maxheap = []
            for num, freq in freq_dic.items():
                heapq.heappush(maxheap, [-freq, -num])
            
            total = 0
            if len(maxheap) < x:
                return sum(temp_num)
            for _ in range(x):
                freq, num = heapq.heappop(maxheap)
                total += freq * num
            return total

        res = []
        n = len(nums)
        res_len = n - k + 1   
        for idx in range(n - k + 1):
            left, right = idx, idx + k
            total = findFreq(left, right)
            res.append(total)
        
        return res
