## Method
### Approach
1. Find the mid
2. Reverse the mid to the end
3. Compare
#### Complexity:
- Time: O(n)
- Space: O(1)
### Code
```python
class Solution:
    def isPalindrome(self, head: Optional[ListNode]) -> bool:
	'''
	1. Find the mid
	2. Reverse the mid to the end
	3. Compare
	   O(n) O(1)
	'''
        def _findhalf(node):
            fast = slow = node
            while fast.next and fast.next.next:
                slow = slow.next
                fast = fast.next.next
            
            return slow

        def _reverse(node):
            cur = node
            prev = None
            while cur:
               nxt = cur.next
               cur.next = prev
               prev = cur
               cur = nxt
            return prev

        if not head or not head.next:
            return True

        half_of_head = _findhalf(head)
        reverse_half = _reverse(half_of_head.next)
        
        p1, p2 = head, reverse_half
        while p2:
            if p1.val != p2.val:
                return False
            p1 = p1.next
            p2 = p2.next
        
        half_of_head.next = _reverse(reverse_half)
        return True
```
