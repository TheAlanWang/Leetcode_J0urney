### Complexity
- Time complexity: O(n)
- Space complexity: O(n)
```python
class Solution:
    def reorderList(self, head: Optional[ListNode]) -> None:
        """
        Do not return anything, modify head in-place instead.
        """
        def _findhalf(node):
            slow = fast = node
            while fast and fast.next:
                fast = fast.next.next
                slow = slow.next
            
            return slow

        def _reverse(node):
            cur = node
            prev = None
            while cur:
                nxt = cur.next
                cur.next = prev
                prev = cur
                cur = nxt
            
            return prev
        
        half_start = _findhalf(head) # right side of mid if even
        reverse = _reverse(half_start.next)
        half_start.next = None
        
        first = head
        second = reverse
        while second:
            t1 = first.next
            t2 = second.next
            
            first.next = second
            second.next = t1 # reconnects the first half to its original chain

            first = t1
            second = t2
```
